```tsx

  return (
    <html lang="en">
      <head />
      <body>

    </html>

```

## Usage

```tsx
import { showTxSonner } from "@/components/token-kit/token-tx-sonner";
```

```tsx
showTxSonner(
  "0x6ab620baa39d4bf784a229c1aa4df0b25267863a4e623785d8937c48c6b4170d",
  "https://etherscan.io/tx/",
  5000,
);
```
