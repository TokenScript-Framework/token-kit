## Usage

```tsx
import { TokenThumbnail } from "@/registry/default/ui/token-thumbnail";
```

```tsx
<TokenThumbnail
  token={{
    address: "0xdac17f958d2ee523a2206206994597c13d831ec7",
    name: "Tether USD",
    chainId: 1,
    logoURI:
      "https://assets.coingecko.com/coins/images/325/standard/Tether.png",
    verified: true,
  }}
/>
```
